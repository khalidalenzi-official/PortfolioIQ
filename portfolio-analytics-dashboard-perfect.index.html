const API_KEY = 'YOUR_KEY_HERE'; const ticker = '2222.SR'; // Saudi Aramco const url = `https://www.alphavantage.co/query? function=GLOBAL_QUOTE &symbol=${ticker} &apikey=${API_KEY}`; fetch(url) .then(r => r.json()) .then(data => { const price = data['Global Quote']['05. price']; console.log('Aramco price: SAR', price); });
